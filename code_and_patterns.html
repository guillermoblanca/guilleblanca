Singleton

/* C# Singleton */
public class Singleton {
    private static Singleton instance;
    private Singleton() { }
    public static Singleton Instance {
        get {
            if(instance == null) instance = new Singleton();
            return instance;
        }
    }
}
# Python Singleton
class Singleton:
    _instance = None
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super(Singleton, cls).__new__(cls)
        return cls._instance
// C++ Singleton
class Singleton {
private:
    static Singleton* instance;
    Singleton() {}
public:
    static Singleton* getInstance() {
        if(!instance)
            instance = new Singleton();
        return instance;
    }
};

Patrón creacional que asegura que una clase tenga una sola instancia y proporciona un punto de acceso global.
